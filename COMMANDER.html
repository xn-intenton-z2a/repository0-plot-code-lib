Commander.js CLI Framework

Date Retrieved: 2023-11-29

Installation

Install via npm:

npm install commander

Usage

const { Command } = require('commander'); const program = new Command();

program.version('1.0.0', '-v, --version', 'output the version number');

program.command('install [options]') .alias('i') .description('install a package') .option('-D, --save-dev', 'save to devDependencies') .action((pkg, options) => { console.log(installing ${pkg} to ${options.saveDev ? 'devDependencies' : 'dependencies'}); });

program.parse(process.argv);
